import { useRouter } from "next/router";
import InternalLayout from "@/components/layouts/InternalLayout";
import Weeklies from "@/components/elements/Weekly/Weeklies";
import Pagination from "@/components/elements/General/Pagination";
import content from "./posts.json";

export const meta = {
  title: "Weekly",
  description: "Listagem de todas as edições da Weekly do RPG do Mestre",
  color: "blue",
};

export default function Weekly() {
  const router = useRouter();
  const { page } = router.query;
  return (
    <InternalLayout meta={meta}>
      <div className="flex flex-wrap sm:-m-12">
        <Weeklies data={content.data} page={page ?? 1} perPage={10} />
      </div>
      <Pagination
        page={page ?? 1}
        perPage={10}
        itemsCount={content.data.length}
        href="/weekly"
      />
    </InternalLayout>
  );
}
