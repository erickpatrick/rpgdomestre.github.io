import { useRouter } from "next/router";
import InternalLayout from "@/components/layouts/InternalLayout";
import Articles from "@/components/elements/Article/Articles";
import Pagination from "@/components/elements/General/Pagination";
import content from "./posts.json";

export const meta = {
  title: "Artigos",
  description: "Listagem de todos os artigos do RPG do Mestre",
  color: "pink",
  keywords:
    "rpg, roleplaing game, rpg do mestre, rpgdm, jogos de interpretação de papeis, interpretação, artigos, blog, artigos sobre rpg",
};

export default function Weekly() {
  const router = useRouter();
  const { page } = router.query;
  return (
    <InternalLayout meta={meta}>
      <div className="flex flex-wrap sm:-m-12">
        <Articles data={content.data} page={page ?? 1} perPage={10} />
      </div>
      <Pagination
        page={page ?? 1}
        perPage={10}
        itemsCount={content.data.length}
        href="/artigos"
      />
    </InternalLayout>
  );
}
